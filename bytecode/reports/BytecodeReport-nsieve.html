def countSieve(m, primes) {
    def i, k
    def count = 0

    i = 2
    while (i <= m) {
        primes[i] = true
        i++
    }

    i = 2
    while (i <= m) {
        if (primes[i]) {
            k = i + i
            while (k <= m) {
                primes[k] = false
                k += i
            }
            count++
        }
        i++
    }
    return count
}

def padNumber(number, fieldLen) {
    def bareNumber = "" + number
    def numSpaces = fieldLen - bareNumber.length()
    def sb = new StringBuffer(' ' * numSpaces)
    sb.append(bareNumber)
    return sb.toString()
}
